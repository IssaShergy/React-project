//useEffect: Allows you to perform side effects in functional components, such as fetching data or subscribing to events.
example:
React.useEffect(() => {
            // Fetch data from an API
            fetch('https://api.example.com/data')
                .then(response => response.json())
                .then(data => {
                    // Update state with the fetched data
                    setData(data);
                });
        }, []); // Empty dependency array means this effect runs only once after the initial render.
//useContext: Allows you to access the context value in functional components.
example:
const MyContext = React.createContext();
        const MyComponent = () => {
            const contextValue = React.useContext(MyContext);
            return <div>{contextValue}</div>;
        };
//useReducer: Allows you to manage complex state logic in functional components using a reducer function.
example:
const initialState = { count: 0 };
        const reducer = (state, action) => {
            switch (action.type) {
                case 'increment':
                    return { count: state.count + 1 };
                case 'decrement':
                    return { count: state.count - 1 };
                default:
                    return state;
            }
        };
        const MyComponent = () => {
            const [state, dispatch] = React.useReducer(reducer, initialState);
            return (
                <div>
                    <p>Count: {state.count}</p>
                    <button onClick={() => dispatch({ type: 'increment' })}>Increment</button>
                    <button onClick={() => dispatch({ type: 'decrement' })}>Decrement</button>
                </div>
            );
        };

//package-lock.json
يتم إنشاء ملف package-lock.json تلقائيًا عند تثبيت الحزم باستخدام npm. يحتوي هذا الملف على معلومات حول الحزم المثبتة وإصداراتها، ويستخدم لضمان أن يتم تثبيت نفس الإصدارات في كل مرة يتم فيها تشغيل npm install. يساعد هذا في الحفاظ على استقرار المشروع وتجنب المشاكل الناتجة عن تحديثات الحزم غير المتوافقة.
//package.json
يحتوي ملف package.json على معلومات حول المشروع، مثل اسم المشروع، الإصدار، الوصف، المؤلف، والاعتمادات (dependencies) التي يحتاجها المشروع للعمل. يتم استخدام هذا الملف لإدارة الحزم والمكتبات التي يعتمد عليها المشروع، ويمكن أيضًا تحديد سكريبتات (scripts) لتشغيل أو بناء المشروع بسهولة.
//export
تستخدم الكلمة المفتاحية export لتصدير المتغيرات أو الدوال أو الكائنات من ملف JavaScript بحيث يمكن استيرادها واستخدامها في ملفات أخرى. هناك نوعان رئيسيان من التصدير: التصدير الافتراضي (default export) والتصدير المسمى (named export).
